The four operators are "add", "subtract", "divide", "multiply".

MY SOLUTION: 

        const arithmetic = (a, b, operator) => {

            case "multiply": return a * b;
            case "divide": return a / b;
            default: throw new RangeError(`Not supported operator '${operator}'`);
          }
        }
